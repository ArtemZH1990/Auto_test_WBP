import time

from selenium import webdriver


browser = webdriver.Chrome()
browser.set_window_size(1920,1080)
browser.get('https://qatest-dev.indvp.com/')
time.sleep(7)
browser.set_window_size(1600,900)
time.sleep(7)
browser.set_window_size(1280,960)
time.sleep(7)
browser.set_window_size(800,600)
time.sleep(7)
browser.set_window_size(350,700)
time.sleep(7)

browser.quit()


Здесь проверку накидал, (не под себя, хоп-хэй) минимальную, по прохождению разрешения. Python+selenium. Если нужно проверить не наслаиваются ли элементы, может пропадают.
